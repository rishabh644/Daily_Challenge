{

        QueueNode *temp=new QueueNode(x);

//Function to pop front element from the queue.
int MyQueue :: pop()
{
        // Your Code
        
        if(front==NULL)
        return -1;
        
        int val=front->data;
        front=front->next;
        
        return val;
        
        
}
